&   5 x^2 - 2 x + 1 \\
&	D = b^2 - 4ac \\
&	D = 2^2 - 4*5*1 \\
&	D = -16 \\
&	D < 0 => Complex_Roots \\
& 	x1 = \frac{-b + i{\sqrt{-D}}}{2a} \\
& 	x2 = \frac{-b - i{\sqrt{-D}}}{2a} \\
&	x1 = \frac{2 + i{\sqrt{16}}}{10} \\
& 	x2 = \frac{2 - i{\sqrt{16}}}{10} 
\end{align*}
