\begin{minted}{js}
var n = 0;
var c = 8;

var start = 0;

function setup() {
  createCanvas(1200, 650);
  angleMode(DEGREES);
  colorMode(HSB);
}

function draw() {
  background(0);
  translate(width / 2, height / 2);
  rotate(n * 0.3);
  for (var i = 0; i < n; i++) {
    var a = i*(random(0,360)) * (random(0,360));
    var r = c * tan(sqrt(i))*random(25,50);
    var x = r * cos(a);
    var y = r * sin(a);
    var hu = (sin(tan(start + i * 0.5)));
    hu = map(hu, -1, 1, 0, 320);
    fill(hu, 255, 255);
    noStroke();
    ellipse(x, y, 4, 4);
  }
  n += 5;
  start += 5;
}
\end{minted}
